\newcommand{\abgloopaux}[3]{%
	\ifnumcomp{#2}{>}{#3}{%
	}{%
		\StrChar{#1}{#2}[\thischar]%
		\StrChar{#1}{\numexpr#2+1\relax}[\nextchar]%
		\IfEq{\nextchar}{;}{%
			\abgmapcompose{\thischar}{;}%
			\abgloopaux{#1}{\numexpr#2+2\relax}{#3}%
		}{%
			\abgmapchar{\thischar}%
			\abgloopaux{#1}{\numexpr#2+1\relax}{#3}%
		}%
	}%
}
